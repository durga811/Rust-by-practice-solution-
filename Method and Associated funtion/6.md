#[derive(Debug)]
enum TrafficLightColor {
    Red,
    Yellow,
    Green,
}

// Implement TrafficLightColor with a method.

    fn color(&self)->&str{
        match self {
         TrafficLightColor::Yellow=>"yellow",
         _=>"something wrong happened in match",
        }
    }
}

fn main() {
    let c = TrafficLightColor::Yellow;

    assert_eq!(c.color(), "yellow");
